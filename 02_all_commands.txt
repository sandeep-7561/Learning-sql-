Create database:
CERATE DATABASE {name of database);


Select from database:
USE {name of database);


Create Table in database:
CREATE TABLE students (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    age INT
);


insert data into table in sql:
INSERT INTO students(name, age)
VALUES("Sandeep", 20);

multub values:
INSERT INTO students(name, age)
VALUES
("Amit",19),
("Rohan",21),
("Priya",20);

if you want to see all table:
Select * from students;

if you want ot see just some columns:
select name from students;


user of where in sql:
select {col name} from {table name} WHERE age=20; /condition like constructors


examples: (queries)
SELECT * FROM students;
SELECT * FROM students WHERE age = 20;
SELECT * FROM students WHERE age = 20;
SELECT * FROM students WHERE age < 20;
SELECT * FROM students WHERE name = "Priya";



conditons in where:
SELECT *
FROM students
WHERE condition1 AND condition2;

use also (or,and,not)
where not age = 20;
real life example:
SELECT *
FROM users
WHERE email="abc@gmail.com"
AND password="123456";

and or or sathme:
SELECT *
FROM students
WHERE (name="Sandeep" OR age=20)
AND id=5;






UPDATE use in sql:
UPDATE students
SET age = 21
WHERE id = 1;

DELETE row(records):
DELETE FROM students
WHERE id = 2;


Delete all records from table:
TRUNCATE TABLE students;

Delete table (danger):
DROP TABLE students;

Delete database (danger):
DROP DATABASE school;




ORDER = Sequence / Kram:
SELECT *
FROM students
ORDER BY age;

same order by but using ASC (Ascending):
Small → Large
A → Z
0 → 9
SELECT *
FROM students
ORDER BY name/age ASC;

DESC (Descending):
SELECT *
FROM students
ORDER BY age DESC;

limit (if you want to see top 3 rows):
SELECT *
FROM students
LIMIT 3;

Order by + limit:
SELECT *
FROM students
ORDER BY age DESC
LIMIT 2;
