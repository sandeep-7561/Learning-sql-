Create database:
CERATE DATABASE {name of database);


Select from database:
USE {name of database);


Create Table in database:
CREATE TABLE students (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    age INT
);


insert data into table in sql:
INSERT INTO students(name, age)
VALUES("Sandeep", 20);

DESC (Descending):
SELECT *
FROM students

limit (if you want to see top 3 rows):
SELECT *
FROM students
LIMIT 3;

Order by + limit:

FROM students
ORDER BY age DESC

(WHERE name LIKE "%ep"; (%ep means end the name with ep))
