
Data Analysis:
Aggregate Functions
COUNT()
SUM()
AVG()
MIN()
MAX()


COUNT(): (how much rows are there including null)
SELECT COUNT(*)
FROM students;

(count all age rows but not including null value)
SELECT COUNT(age)
FROM students;

(Count only null value in age)
select count(*) from students where age iS NULL;


SUM():
SELECT SUM(salary)
FROM employees;


AVG():
SELECT AVG(age)
FROM students;

MIN():
SELECT MIN(age)
FROM students;

MAX():
SELECT MAX(age)
FROM students;

Aggregate + WHERE:
SELECT COUNT(*)
FROM students
WHERE age >=20;

-----------------------------------------------

GROUP BY:
SELECT city, Count(*) ( show city name and total students form this city)
FROM students
GROUP BY city;

HAVING:
WHERE (filter row)
HAVING (filter group)

SELECT city, COUNT(*)
FROM students
GROUP BY city
HAVING COUNT(*) >= 2;

real world examples:
(amazon)
SELECT category,
COUNT(*)
FROM products
GROUP BY category;


(Hospital)
SELECT blood_group,
COUNT(*)
FROM patients
GROUP BY blood_group;
